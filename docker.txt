#Docker Compose ist ein Tool, mit dem du mehrere Container gleichzeitig konfigurieren und starten kannst.
#Statt jeden Container manuell zu starten, schreibst du alles in einer YAML-Datei (docker-compose.yml),
#und Docker Compose baut und startet alles automatisch.


docker-compose.yml ---> file erstellen

version: "3"
services:
    api:                                                #api ist name des service , ein service ist ein container , ein image ist eine klasse" bauplan" und ein container ist ein objekt davon
        build: .                                        # baut ein image aus dem dockerfile im root directory
        ports:
            - 8000:8000                                             #<port on localhost>:<port on container> , wenn ich eine anfrage an meinen container sende über port 4000 wird er an port 8000 weitergeleitet, hier auf 8000 erstmal lassen




docker-compose up - d    # baut die images und startet container
docker ps -a             # zeigt alle container an
